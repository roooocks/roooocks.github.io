## 요약
배열을 기반으로 구현된 리스트
<br>


## 특징
1. 배열 길이가 초기에 결정되어야 한다. 변경은 불가능하다.
2. 데이터 삭제 과정에서 데이터의 이동(복사)가 매우 빈번히 일어난다.
3. 데이터 참조가 쉽고! 빠르다! 인덱스 값을 기준으로 어디든 한번에 참조 가능하다.

## 코드
```c
#ifndef __ARRAY_LIST_H__
#define __ARRAY_LIST_H__

#define TRUE		1
#define FALSE		0

#define LIST_LEN	100		// 리스트의 전체 길이

typedef int LData;

typedef struct __ArrayList {
    LData arr[LIST_LEN];	// 리스트의 저장 공간(배열)
    int len;				// 리스트의 현재 길이
    int curPos;				// 현재 데이터 참조 위치
} ArrayList;

// 순차 리스트 -> 연결 리스트가 되었을 때, 아래 ArrayList만 바꿔주면 된다.
typedef struct __ArrayList List;

void ListInit(List* plist);
void LInsert(List* plist, LData data);

int LFirst(List* plist, LData* pdata);
int LNext(List* plist, LData* pdata);

LData LRemove(List* plist);
int LCount(List* plist);

#endif
```

```c
#include <stdio.h>
#include "ArrayList.h"

void ListInit(List* plist)
{
    plist->len = 0;		// 현재 개수
    plist->curPos = -1;	// 현재 가르키는 위치. 0을 넣으면 첫번째를 가리키기 때문에 -1
}

void LInsert(List* plist, LData data)
{
    if (plist->len >= LIST_LEN)
    {
        printf("저장이 불가능합니다.\n");
        return;
    }

    plist->arr[plist->len] = data;
    plist->len += 1;
}

int LFirst(List* plist, LData* pdata)
{
    if (plist->len == 0)
        return FALSE;

    plist->curPos = 0;
    *pdata = plist->arr[0];

    return TRUE;
}

int LNext(List* plist, LData* pdata)
{
    if (plist->curPos >= (plist->len - 1))
        return FALSE;

    plist->curPos += 1;
    *pdata = plist->arr[plist->curPos];

    return TRUE;
}

LData LRemove(List* plist)
{
    int i;
    LData temp = plist->arr[plist->curPos];		// 삭제해야 할 데이터 임시 저장

    for (i = plist->curPos; i < (plist->len) - 1; i++)
        plist->arr[i] = plist->arr[i+1];

    plist->len -= 1;
    plist->curPos -= 1;

    return temp;
}

int LCount(List* plist)
{
    return plist->len;
}
```

```c
#include <stdio.h>
#include "List/ArrayList.h"

int main()
{
    List list;
    int data;
	
    // 리스트 : 초기화
    ListInit(&list);

    // 리스트 : 데이터 저장
    LInsert(&list, 11); LInsert(&list, 22);
    LInsert(&list, 33); LInsert(&list, 44);
    LInsert(&list, 55);

    // 리스트 : 개수 출력
    printf("현재 리스트 개수 : %d\n", LCount(&list));

    // 리스트 : 전체 데이터 출력
    if (LFirst(&list, &data))
    {
    	printf("%d ", data);

        while (LNext(&list, &data))
	        printf("%d ", data);
    }
    printf("\n\n");

    // 리스트 : 숫자 22 삭제
    if (LFirst(&list, &data))
    {
    	if (data == 22)
	        LRemove(&list);

        while (LNext(&list, &data))
        {
            if (data == 22)
                LRemove(&list);
        }
    }

    // 리스트 : 삭제 후 남은 데이터 전체 출력
    if (LFirst(&list, &data))
    {
        printf("%d ", data);

        while (LNext(&list, &data))
            printf("%d ", data);
    }
    printf("\n\n");

    return 0;
}
```
