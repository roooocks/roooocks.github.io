```
# 원격 저장소 확인
git remote

# 원격 저장소 URL 확인 (fetch, push URL 확인)
git remote -v

# 새로운 원격 저장소 추가
git remote add upstream https://github.com/user/repository.git

# 원격 저장소 삭제
git remote rm upstream

# 원격 저장소 이름 변경
git remote rename upstream main
```
